013public class Stations
014{
015    ArrayList<String> staList = new ArrayList<String>();
016    ArrayList<String> staIndex = new ArrayList<String>();
017    String strFilePath;
018
019    /**
020     * Constructor
021     * constructor that will accept a string representing the full path to the FBIN.txt file
022     * store it in an instance variable called strFilePath
023     * read the file record by record and add each station weather string to the staList
024     * store the corresponding station ID to the staIndex array list.
025     * @param strVar - a string representing the full path to the FBIN.txt file
026     * @throws Exception
027     */
028    public Stations(String strVar) throws Exception
029    {
030        //declare variable
031        strFilePath=strVar;
032        String strRecord;
033        File myFile=new File(strFilePath);
034        if(myFile.exists())
035        {
036            Scanner inputFile= new Scanner(myFile);
037            int intCount=0;
038            while(inputFile.hasNext())
039            {
040                strRecord= inputFile.nextLine();
041                //skip line
042                if(intCount>6)
043                {
044                    //store the station weather in staList
045                    staList.add(strRecord);
046                    //store the first three characters in staIndex
047                    staIndex.add(strRecord.substring(0,3));
048                }
049                intCount++;
050            }
051            inputFile.close();
052        }
053    }
054
055    /**
056     * length() function that will return the number of stations stored in the arraylist
057     * @return - the number of stations stored in the arraylist
058     */
059    public int length()
060    {
061        return staList.size();
062    }
063
064    /**
065     * a getStaWea(strStationID) function that will accept the station id as a parameter and return the full station weather as a string
066     * @param strStaID - the station id
067     * @return - the full station weather as a string
068     */
069    public String getStaWea(String strStaID)
070    {
071
072        String strRet="N/A";
073        if(exists(strStaID)) strRet=staList.get(staIndex.indexOf(strStaID));
074        return strRet;
075    }
076
077    /**
078     * exists(strStationID) function that checks to see if the station ID exists in the staIndex arraylist
079     * @param strStaID - the station ID
080     * @return- boolean
081     */
082    public Boolean exists(String strStaID)
083    {
084        Boolean blnRet=false;
085        int intPos=0;
086        String strRet;
087        //look up the station Id in staIndex
088        intPos=staIndex.indexOf(strStaID);
089        //If indexOf return a -1 it did not find it
090        if(intPos<0)
091        {
092            blnRet=false;
093        }else
094        {
095            blnRet= true;
096        }
097        return blnRet;
098    }
099}
